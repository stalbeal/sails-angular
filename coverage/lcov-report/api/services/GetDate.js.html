exports.hour = function(date) {
    var hour = date.getHours();
    var minutes = date.getMinutes();
 
    var hourResult = "";
    if (minutes < 10 && hour < 10) {
        hourResult = "0" + hour + ":0" + minutes;
    } else {
        hourResult = hour + ":" + minutes;
    }
    return hourResult;
 
};
